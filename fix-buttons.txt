브라우저 콘솔에서 다음 코드를 실행하여 일괄 승인/거부 버튼을 수정하세요:

// 1. 기존 이벤트 리스너 제거
document.getElementById('bulkApprove').removeEventListener('click', window.onClick);
document.getElementById('bulkReject').removeEventListener('click', window.onClick);

// 2. 새로운 일괄 승인 함수
function newBulkApprove() {
    console.log('일괄 승인 버튼 클릭됨');
    
    if (!window.dataManager) {
        console.error('dataManager가 없습니다!');
        return;
    }
    
    const allPendingRequests = window.dataManager.leaveRequests.filter(req => req.status === 'pending');
    console.log('모든 대기중인 요청:', allPendingRequests);
    
    if(allPendingRequests.length === 0) {
        alert('처리할 대기중인 요청이 없습니다.');
        return;
    }
    
    if(confirm(`모든 대기중인 ${allPendingRequests.length}개 요청을 승인하시겠습니까?`)){
        allPendingRequests.forEach(request => {
            window.dataManager.updateLeaveRequestStatus(request.id, 'approved');
            console.log(`승인됨: ${request.employeeName} (${request.startDate} ~ ${request.endDate})`);
        });
        
        console.log(`총 ${allPendingRequests.length}개 요청이 승인되었습니다.`);
        
        // 페이지 새로고침
        setTimeout(() => location.reload(), 1000);
    }
}

// 3. 새로운 일괄 거부 함수
function newBulkReject() {
    console.log('일괄 거부 버튼 클릭됨');
    
    if (!window.dataManager) {
        console.error('dataManager가 없습니다!');
        return;
    }
    
    const allPendingRequests = window.dataManager.leaveRequests.filter(req => req.status === 'pending');
    console.log('모든 대기중인 요청:', allPendingRequests);
    
    if(allPendingRequests.length === 0) {
        alert('처리할 대기중인 요청이 없습니다.');
        return;
    }
    
    if(confirm(`모든 대기중인 ${allPendingRequests.length}개 요청을 거부하시겠습니까?`)){
        allPendingRequests.forEach(request => {
            window.dataManager.updateLeaveRequestStatus(request.id, 'rejected');
            console.log(`거부됨: ${request.employeeName} (${request.startDate} ~ ${request.endDate})`);
        });
        
        console.log(`총 ${allPendingRequests.length}개 요청이 거부되었습니다.`);
        
        // 페이지 새로고침
        setTimeout(() => location.reload(), 1000);
    }
}

// 4. 새로운 이벤트 리스너 등록
document.getElementById('bulkApprove').addEventListener('click', newBulkApprove);
document.getElementById('bulkReject').addEventListener('click', newBulkReject);

console.log('일괄 승인/거부 버튼이 수정되었습니다!');
console.log('이제 버튼을 클릭해보세요!');
