브라우저 콘솔에서 다음 코드를 실행하여 테스트 데이터를 대기중 상태로 초기화하세요:

// 1. 모든 연차 신청을 대기중 상태로 변경
function resetToPending() {
    if (!window.dataManager) {
        console.error('dataManager가 없습니다!');
        return;
    }
    
    console.log('현재 연차 신청 상태:', window.dataManager.leaveRequests);
    
    // 모든 연차 신청을 pending 상태로 변경
    window.dataManager.leaveRequests.forEach(request => {
        request.status = 'pending';
        delete request.approvedDate;
        delete request.approvedBy;
    });
    
    // 데이터 저장
    window.dataManager.saveData('leaveRequests', window.dataManager.leaveRequests);
    
    console.log('모든 연차 신청이 대기중 상태로 변경되었습니다!');
    console.log('변경된 상태:', window.dataManager.leaveRequests);
    
    // 페이지 새로고침
    setTimeout(() => location.reload(), 1000);
}

// 2. 실행
resetToPending();
